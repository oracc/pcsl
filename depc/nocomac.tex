\newbox\nocobox \newbox\nocoheaderbox \newbox\nocoheadertmp
\newdimen\nocorowht

\def\nchbox#1{\hbox to.75in{\nine\rm#1\hss}}
\def\nchboxmus#1{\hbox to1in{\nine\rm#1\hss}}
\def\nchboxper#1{\hbox to.6in{\nine\rm#1\hss}}
\def\nchboxm#1{\hbox to2.4in{\nine\rm#1\hfil}}

\newcount\ltfirstpage

\def\beginnocotab{%\newpage
  \bgroup\offinterlineskip \lineskip1pt
  \ltfirstpage\the\pageno
  \setbox\nocoheaderbox=\vbox{\vfuzz12pt%
  \nocorowx{\hbox{CDLI}}%
           {\nchbox{MusNo}}%
           {\nchbox{Time}}%
           {\nchbox{Place}}%
           {\nchbox{Pub}}%
           {\nchboxm{\quad Missing}}}%
   %\righthdrinfo{}%
   %\headersep20pt
   \copy\nocoheaderbox
}

\def\nocoheader{%
    \ifnum\pageno>\ltfirstpage
      \lefthdrinfo{}%
      \righthdrinfo{}%
      \headersep20pt
      \setbox\nocoheadertmp=\vbox{%
        \kern-12pt
        \hbox to\hsize{\eight\it\the\runheadtoks\hfil\Sans\nine\rm\folio}%
        \kern2pt
        \unvcopy\nocoheaderbox
      }%
      \centrehdrinfo{\box\nocoheadertmp}%
    \fi}

\def\endnocotab{\nocoheader\vfil\supereject\egroup}

\def\nocorowx#1#2#3#4#5#6{%
     \hrule height1.5pt
     \hbox to\hsize{%
            \vbox to\nocorowht{\vfil\nchbox{#1}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchboxmus{#2}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchboxper{#3}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchbox{#4}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchbox{#5}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchboxm{#6}\vfil}%
     \hfil
   }%
   \kern12pt
   \hrule height1.5pt
}

\def\nocorow#1#2#3#4#5#6{%
   \nocoheader
   \setbox\nocobox\vbox{\hsize2in#6}%
   \nocorowht=\ht\nocobox
   \advance\nocorowht2pt % compensate for lineskips
   \hbox to\hsize{%
            \vbox to\nocorowht{\vfil\nchbox{#1}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchboxmus{#2}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchboxper{#3}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchbox{#4}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchbox{#5}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchboxm{#6}\vfil}%
     \hfil
   }%
   \kern2pt
   \hrule height.4pt
   \vskip3pt plus1pt minus1pt
}
