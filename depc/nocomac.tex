\newbox\nocobox \newbox\nocoheaderbox \newdimen\nocorowht

\def\nchbox#1{\hbox to.75in{\small#1\hss}}
\def\nchboxmus#1{\hbox to1in{\small#1\hss}}
\def\nchboxper#1{\hbox to.6in{\small#1\hss}}
\def\nchboxm#1{\hbox to2.4in{\small#1\hss}}

\def\usenocoheaderbox{\copy\nocoheaderbox}

\def\nocotabinit{%
  \global\setbox\nocoheaderbox=\vbox{%
  \nocorowx{\hbox{CDLI}}%
           {\nchbox{MusNo}}%
           {\nchbox{Time}}%
           {\nchbox{Place}}%
           {\nchbox{Pub}}%
           {\nchboxm{\quad Missing}}}%
}
\def\nocorowx#1#2#3#4#5#6{%
     \hrule height1.5pt width6.5in
     \kern1pt
     \hbox to\hsize{%
            \vbox to\pcslhdrht{\vfil\nchbox{#1}\vfil}%
     \nquadb\vbox to\pcslhdrht{\vfil\nchboxmus{#2}\vfil}%
     \nquadb\vbox to\pcslhdrht{\vfil\nchboxper{#3}\vfil}%
     \nquadb\vbox to\pcslhdrht{\vfil\nchbox{#4}\vfil}%
     \nquadb\vbox to\pcslhdrht{\vfil\nchbox{#5}\vfil}%
     \nquadb\vbox to\pcslhdrht{\vfil\nchboxm{#6}\vfil}%
     \hss
   }%
   \kern1pt
   \hrule height1.5pt width6.5in
}

\def\beginnocotab{
  \bgroup\offinterlineskip \lineskip1pt
  \ltfirstpage\value{page}%
  \copy\nocoheaderbox
  \let\currheaderbox\usenocoheaderbox
}

\def\endnocotab{\clearpage\egroup}

\def\nocorow#1#2#3#4#5#6{%
   \setbox\nocobox\vbox{\hsize2in#6}%
   \nocorowht=\ht\nocobox
   \advance\nocorowht2pt % compensate for lineskips
   \hbox to\hsize{%
            \vbox to\nocorowht{\vfil\nchbox{#1}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchboxmus{#2}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchboxper{#3}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchbox{#4}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchbox{#5}\vfil}%
     \nquadb\vbox to\nocorowht{\vfil\nchboxm{#6}\vfil}%
     \hfil
   }%
   \kern2pt
   \hrule height.4pt
   \vskip3pt plus1pt minus1pt
}
