\def\sqhboxnH#1{\hbox to2.3in{\footnotesize#1\hfil}}
\def\sqhboxpH#1{\hbox to1.2in{\footnotesize#1\hfil}}
\def\sqhboxLH#1{\hbox to3in{\footnotesize#1\hfil}}
\def\sqhboxn#1{\hbox to2.3in{\scriptsize#1\hfil}}
\def\sqhboxd#1{\hbox to2in{\quad#1\hfil}}
\def\sqhboxp#1{\hbox to1.2in{\small#1\hfil}}
\def\sqhboxL#1{\hbox to3in{\small#1\hfil}}
\def\sqhboxl#1{\hbox to3in{\scriptsize#1\hfil}}

\def\useseqdbheaderbox{\copy\seqdbheaderbox}

\def\seqdbtabinit{%
  \setbox\seqdbheaderbox=\vbox{\vfuzz12pt%
     \seqdbrowx{CHARNAME/GLYFNAME/DECOMP}%
               {PUA}%
               {LIG/.liga}%
   }%
}
\def\seqdbrowx#1#2#3{%
   \hrule height1.5pt width6.5in
   \kern1pt
   \hbox to\hsize{%
            \vbox to\pcslhdrht{\vfil\sqhboxnH{\rm#1}\vfil}%
     \nquadb\vbox to\pcslhdrht{\vfil\sqhboxpH{\rm#2}\vfil}%
     \nquadb\vbox to\pcslhdrht{\vfil\sqhboxLH{\rm#3}\vfil}%
     \hss
   }%
   \kern1pt
   \hrule height1.5pt width6.5in
}

\def\beginseqdb{%\newpage
  \bgroup\offinterlineskip \lineskip1pt
  \ltfirstpage\value{page}%
  \copy\seqdbheaderbox
  \let\currheaderbox\useseqdbheaderbox
}

\def\endseqdb{\clearpage\egroup}
